-- sqlite> .tables
-- sqlite> .schema jobs
-- sqlite> .schema spideredPages

CREATE TABLE "jobs" (
	`uuid`	TEXT NOT NULL,
	`date`	TEXT NOT NULL,
	`data`	TEXT NOT NULL,
	PRIMARY KEY(`uuid`)
);

CREATE TABLE `spideredPages` (
	`url`	TEXT NOT NULL,
	`date`	TEXT NOT NULL,
	`data`	TEXT NOT NULL,
	PRIMARY KEY(`url`)
);
